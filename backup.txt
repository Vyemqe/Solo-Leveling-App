#! Tier-Ranking System
def get_rank(level):
    if level >= 300: return "SSS-Rank Hunter"
    if level >= 150: return "SS-Rank Hunter"
    if level >= 100: return "S-Rank Hunter"
    if level >= 75: return "A-Rank Hunter"
    if level >= 50: return "B-Rank Hunter"
    if level >= 25: return "C-Rank Hunter"
    if level >= 10: return "D-Rank Hunter"
    return "E-Rank Hunter"

    "task_id_counter": 4
}

#! Save/Load Data from curr_data.json
def load_data():
    if os.path.exists(SAVE_FILE):
        with open(SAVE_FILE, "r") as f:
            data = json.load(f)
            # Ensure old saves have the new 'archived' key and correct rank
            for task in data.get("tasks", []):
                if "archived" not in task:
                    task["archived"] = False
                if "completed_at" not in task:
                    task["completed_at"] = None
            # Get the current rank
            data["player"]["title"] = get_rank(data["player"].get("level", 1))
            return data
    return DEFAULT_DATA.copy()

def save_data(data):
    with open(SAVE_FILE, "w") as f:
        json.dump(data, f, indent=4)


#! Load data on start-up

#! Routing API
@app.route("/")
def index():
    # return render_template_string(HTML_TEMPLATE, player=game_data["player"], tasks=game_data["tasks"])
    # Only show tasks that are NOT archived
    active_tasks = [task for task in game_data["tasks"] if not task.get("archived", False)]
    return render_template_string(INDEX_HTML, player=game_data["player"], tasks=active_tasks)

@app.route("/history")
def history():
    # Show ALL completed tasks (whether archived or recently completed)
    completed_tasks = [task for task in reversed(game_data["tasks"]) if task["completed"]]
    return render_template_string(HISTORY_HTML, tasks=completed_tasks)

@app.route("/complete", methods=["POST"])
def complete_task():
    task_id = request.json.get("id")
    leveled_up = False
    for task in game_data["tasks"]:
        if task["id"] == task_id and not task["completed"]:
            task["completed"] = True
            # Stamp the exact time the task was completed
            task["completed_at"] = datetime.now().strftime("%Y-%m-%d %I:%M %p")
            game_data["player"]["xp"] += task["xp"]
            # Level up loop (in case a task gives enough XP to jump multiple levels)
            while game_data["player"]["xp"] >= game_data["player"]["max_xp"]:
                game_data["player"]["level"] += 1
                game_data["player"]["xp"] -= game_data["player"]["max_xp"]
                game_data["player"]["max_xp"] = int(game_data["player"]["max_xp"] * 1.5)
                leveled_up = True
            # Update rank based on new level
            game_data["player"]["title"] = get_rank(game_data["player"]["level"])
            # Save game
            save_data(game_data)
            return jsonify({"success": True, "player": game_data["player"], "leveled_up": leveled_up})
    return jsonify({"success": False}), 400

@app.route("/add", methods=["POST"])
def add_task():
    title = request.json.get("title")
    xp = request.json.get("xp", 40)
    game_data["tasks"].append({
        "id": game_data["task_id_counter"],
